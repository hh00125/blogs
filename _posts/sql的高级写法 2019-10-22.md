## 给斗鱼主播小姐姐点赞数目，按月分组取得前三，可以并列前三哦！
```SQL
-- 方案1
SELECT 
rate1, user_id1, DAY1, rownum1
FROM
	(
	SELECT
		ml.rate AS rate1,
		ml.user_id AS user_id1,
		ml.DAY AS DAY1,
		@rownum := @rownum + 1 AS rownum1 
	FROM
		(
		SELECT
			count( like_id ) AS rate,
			user_id,
			DATE_FORMAT( create_time, '%Y%m' ) AS DAY 
		FROM
			user_like_record 
		WHERE
			DATE_FORMAT( create_time, '%Y%m' ) LIKE '%2018%' 
		GROUP BY
			user_id,
		DAY 
		ORDER BY
			DAY,
			rate DESC 
		) ml,
		( SELECT @rownum := 0 ) r 
	ORDER BY
		ml.DAY,
		ml.rate DESC 
	) AS vm1
	LEFT JOIN (
	SELECT
		ml.rate AS rate2,
		ml.user_id AS user_id2,
		ml.DAY AS DAY2,
		@rownum := @rownum + 1 AS rownum2 
	FROM
		(
		SELECT
			count( like_id ) AS rate,
			user_id,
			DATE_FORMAT( create_time, '%Y%m' ) AS DAY 
		FROM
			user_like_record 
		WHERE
			DATE_FORMAT( create_time, '%Y%m' ) LIKE '%2018%' 
		GROUP BY
			user_id,
		DAY 
		ORDER BY
			DAY,
			rate DESC 
		) ml,
		( SELECT @rownum := 0 ) r 
	ORDER BY
		ml.DAY,
		ml.rate DESC 
	) AS vm2 ON vm1.DAY1 = vm2.DAY2 
	AND vm1.rate1 < vm2.rate2 
GROUP BY
	vm1.rate1,
	vm1.user_id1,
	vm1.DAY1,
	vm1.rownum1 
HAVING
	count(vm1.rownum1) < 3
ORDER BY DAY1, rate1 DESC;

-- 方案4
SELECT
	t1.groupValue '月份',
	t1.user_id '用户ID',
	t1.score '点赞数量' 
FROM
	( SELECT @rank := 0, @groupValue := '' ) variable,
	(
	SELECT
		@rank :=
	IF
		( @groupValue = monthValue, @rank + 1, 1 ) AS rank,
		@groupValue := monthValue AS groupValue,
		mv.user_id,
		mv.score 
	FROM
		(
		SELECT
			count( 1 ) AS score,
			user_id,
			DATE_FORMAT( create_time, '%Y%m' ) AS monthValue 
		FROM
			user_like_record 
		WHERE
			create_time >= '2018' 
			AND create_time < '2019' 
		GROUP BY
			user_id,
			monthValue 
		) mv 
	ORDER BY
		groupValue,
		score DESC 
	) t1 
WHERE
	t1.rank <= 4;
```
